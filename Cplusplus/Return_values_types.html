Typy wartości zwracanych

C++ pozwala na zwracanie wartości dowolnego typu danych, w tym typów prostych (np. int, float, bool, char) oraz typów złożonych (np. struktury, klasy, wskaźniki). Można także zdefiniować funkcję, która nie zwraca wartości (void), a jedynie wykonuje określone instrukcje.

Funkcja zwracająca strukturę

								
#include <iostream>
using namespace std;

struct Osoba {
  string imie;
  string nazwisko;
  int wiek;
};


Osoba utworz_osobe(string imie, string nazwisko, int wiek) {
  Osoba osoba;
  osoba.imie = imie;
  osoba.nazwisko = nazwisko;
  osoba.wiek = wiek;
  return osoba;
}


int main() {
  Osoba jan = utworz_osobe("Ola", "Kowalska", 25);
  cout << jan.imie << " " << jan.nazwisko << " ma " << jan.wiek << " lat" << endl;
  return 0;
}
Ola Kowalska ma 25 lat
